## 1. 🛠️ 기술 스택

| 분류 | 기술 | 비고 |
|---|---|---|
| Framework | Next.js 14 (App Router) | Server Component 적극 활용 |
| Language | TypeScript | strict mode, Union Type 적극 활용 |

| 상태관리 | React useState / Server Component | 별도 전역 상태 라이브러리 없음 |

export type ETFHolding = {
  id: string
  ticker: string             // 예: 'TIGER 미국배당다우존스'

### ETFPieChart (SVG 직접 구현)

```tsx
// app/portfolio/_components/ETFPieChart.tsx
// recharts ❌ — SVG 직접 구현 ✅
// 이유: 디자인 토큰(CSS Variables) 직접 연동 + 커스텀 인터랙션 자유도

type Slice = ETFHolding & {
  ratio: number
  startAngle: number
  angle: number
}

function polarToCartesian(cx: number, cy: number, r: number, angleDeg: number) {
  const rad = ((angleDeg - 90) * Math.PI) / 180
  return { x: cx + r * Math.cos(rad), y: cy + r * Math.sin(rad) }
}

function describeArc(cx: number, cy: number, r: number, start: number, end: number) {
  const s = polarToCartesian(cx, cy, r, start)
  const e = polarToCartesian(cx, cy, r, end)
  const largeArc = end - start > 180 ? 1 : 0
  return `M ${cx} ${cy} L ${s.x} ${s.y} A ${r} ${r} 0 ${largeArc} 1 ${e.x} ${e.y} Z`
}

// 종목별 색상 — CSS Variables 기반 확장 가능
const SLICE_COLORS = [
  'var(--color-chart-1, #E53E3E)',
  'var(--color-chart-2, #2B6CB0)',
  'var(--color-chart-3, #38A169)',
  'var(--color-chart-4, #D69E2E)',
  'var(--color-chart-5, #805AD5)',
]

export default function ETFPieChart({ holdings }: { holdings: ETFHolding[] }) {
  const [activeIndex, setActiveIndex] = useState<number | null>(null)
  const total = holdings.reduce((sum, h) => sum + h.quantity * h.avgPrice, 0)

  let cumulative = 0
  const slices: Slice[] = holdings.map(h => {
    const ratio = (h.quantity * h.avgPrice) / total
    const slice = { ...h, ratio, startAngle: cumulative, angle: ratio * 360 }
    cumulative += slice.angle
    return slice
  })

  return (
    <div className={styles.wrapper}>
      <svg viewBox="0 0 200 200" className={styles.chart}>
        {slices.map((slice, i) => (
          <path
            key={slice.ticker}
            d={describeArc(100, 100, 90, slice.startAngle, slice.startAngle + slice.angle)}
            fill={SLICE_COLORS[i % SLICE_COLORS.length]}
            opacity={activeIndex === null || activeIndex === i ? 1 : 0.4}
            className={styles.slice}
            onClick={() => setActiveIndex(i === activeIndex ? null : i)}
            aria-label={`${slice.ticker} ${Math.round(slice.ratio * 100)}%`}
          />
        ))}
      </svg>
      {/* 클릭 시 종목 상세 */}
      {activeIndex !== null && (
        <div className={styles.detail}>
          <span className={styles.ticker}>{slices[activeIndex].ticker}</span>
          <span className={styles.ratio}>{Math.round(slices[activeIndex].ratio * 100)}%</span>
        </div>
      )}
    </div>
  )
}
```

// app/dashboard/_components/GoalGauge.tsx
'use client'
import { useState, useEffect } from 'react'
import styles from './GoalGauge.module.scss'

export default function GoalGauge({ percentage }: { percentage: number }) {
  const [width, setWidth] = useState(0)

  useEffect(() => {
    // 마운트 후 한 프레임 뒤 실제 값으로 전환 → CSS transition 발동
    const id = requestAnimationFrame(() => setWidth(Math.min(percentage, 100)))
    return () => cancelAnimationFrame(id)
  }, [percentage])

  return (
    <div className={styles.wrapper}>
      <div className={styles.track} role="progressbar"
        aria-valuenow={percentage} aria-valuemin={0} aria-valuemax={100}>
        <div className={styles.fill} style={{ width: `${width}%` }} />
      </div>
      <span className={styles.label}>{Math.round(percentage)}% 달성</span>
    </div>
  )
}

## 7. 📈 배당 예측 계산 로직

```typescript
// lib/dividend-calculator.ts

export function calcDividendProjection(
  holdings: ETFHolding[],
  yearsToProject: number,
  dailyPurchase: boolean = true
): DividendProjection {
  const tradingDaysPerYear = 250

  const result = holdings.reduce((acc, etf) => {
    const additionalShares = dailyPurchase
      ? tradingDaysPerYear * yearsToProject
      : 0

    const futureQuantity = etf.quantity + additionalShares

    // 배당 성장률 복리 적용
    const futureAnnualDividend =
      etf.annualDividendPerShare *
      Math.pow(1 + etf.dividendGrowthRate, yearsToProject)

    const annualDividend = futureQuantity * futureAnnualDividend

    return {
      monthlyDividend: acc.monthlyDividend + annualDividend / 12,
      annualDividend:  acc.annualDividend  + annualDividend,
      totalShares:     acc.totalShares     + futureQuantity,
    }
  }, { monthlyDividend: 0, annualDividend: 0, totalShares: 0 })

  return { year: yearsToProject as DividendProjection['year'], ...result }
}

// 4개 타임라인 한 번에 계산
export function calcAllProjections(holdings: ETFHolding[]) {
  return [3, 5, 10, 20].map(year => calcDividendProjection(holdings, year))
}
```

폴더 구조에 `middleware.ts`가 없음. 비로그인 사용자가 `/dashboard`, `/portfolio`에 바로 접근 가능.

```typescript
// middleware.ts (루트에 추가 필요)
import { createMiddlewareClient } from '@supabase/auth-helpers-nextjs'
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export async function middleware(req: NextRequest) {
  const res = NextResponse.next()
  const supabase = createMiddlewareClient({ req, res })
  const { data: { session } } = await supabase.auth.getSession()

  const protectedPaths = ['/dashboard', '/portfolio', '/settings']
  const isProtected = protectedPaths.some(p => req.nextUrl.pathname.startsWith(p))

  if (isProtected && !session) {
    return NextResponse.redirect(new URL('/login', req.url))
  }
  return res
}

export const config = {
  matcher: ['/dashboard/:path*', '/portfolio/:path*', '/settings/:path*'],
}
```

#### 4. ETFPieChart — 엣지케이스 NaN 가드

// ❌ 현재 — total=0 시 ratio = 0/0 = NaN
const total = holdings.reduce((sum, h) => sum + h.quantity * h.avgPrice, 0)

// ✅ 수정
export default function ETFPieChart({ holdings }: { holdings: ETFHolding[] }) {
  // 빈 상태 얼리 리턴
  if (holdings.length === 0) {

        <a href="/portfolio">첫 ETF 추가하기 →</a>

  }
